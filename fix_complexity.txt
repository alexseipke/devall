    calculateComplexity(code) {
        // Complejidad ciclom√°tica simplificada
        let complexity = 1;
        
        // Palabras clave que aumentan complejidad
        const patterns = [
            /\bif\b/g,
            /\belse\b/g,
            /\bfor\b/g,
            /\bwhile\b/g,
            /\bcase\b/g,
            /\bcatch\b/g,
            /&&/g,
            /\|\|/g,
            /\?.*:/g
        ];
        
        for (const pattern of patterns) {
            const matches = code.match(pattern);
            if (matches) {
                complexity += matches.length;
            }
        }
        
        return complexity;
    }
